    python
    class Student:
        def init(self, name, age, grade):
            self.name = name
            self.age = age
            self.grade = grade

        def info(self):
            return f"{self.name} - {self.age} жаш, класс: {self.grade}"

        # Жаңы метод
        def study(self):
            print(f"{self.name} азыр окуп жатат!")
   # Объект түзүү
    student1 = Student("Айдана", 15, "9-класс")
    student1.study()
